Golang 标准库使用教程

1. io/io/ioutil 包

1.1 io 包基础

package main

import (
    "fmt"
    "io"
    "os"
)

func main() {
    // 打开文件
    file, err := os.Open("test.txt")
    if err != nil {
        panic(err)
    }
    defer file.Close()

    // 使用io.Copy复制内容
    _, err = io.Copy(os.Stdout, file)
    if err != nil {
        panic(err)
    }

    // 使用io.ReadAll读取全部内容
    file.Seek(0, 0) // 重置文件指针
    content, err := io.ReadAll(file)
    if err != nil {
        panic(err)
    }
    fmt.Println("\nFile content:", string(content))
}

1.2 ioutil 包（已弃用，但了解其功能）

package main

import (
    "fmt"
    "io/ioutil"
    "os"
)

func main() {
    // 读取文件全部内容
    content, err := ioutil.ReadFile("test.txt")
    if err != nil {
        panic(err)
    }
    fmt.Println("File content:", string(content))

    // 写入文件
    err = ioutil.WriteFile("output.txt", []byte("Hello, Go!"), 0644)
    if err != nil {
        panic(err)
    }

    // 读取目录
    files, err := ioutil.ReadDir(".")
    if err != nil {
        panic(err)
    }
    for _, file := range files {
        fmt.Println(file.Name())
    }
}

2. os/exec 包

package main

import (
    "fmt"
    "os/exec"
)

func main() {
    // 执行简单命令
    cmd := exec.Command("ls", "-l")
    output, err := cmd.Output()
    if err != nil {
        panic(err)
    }
    fmt.Println(string(output))

    // 执行命令并获取标准输出和错误
    cmd = exec.Command("go", "version")
    output, err = cmd.CombinedOutput()
    if err != nil {
        panic(err)
    }
    fmt.Println(string(output))

    // 启动长时间运行的进程
    cmd = exec.Command("ping", "google.com")
    cmd.Stdout = os.Stdout
    cmd.Stderr = os.Stderr
    err = cmd.Start()
    if err != nil {
        panic(err)
    }
    fmt.Println("Process started with PID:", cmd.Process.Pid)
    err = cmd.Wait()
    fmt.Println("Process finished with error:", err)
}

3. net/http 包

3.1 HTTP 服务器

package main

import (
    "fmt"
    "net/http"
)

func helloHandler(w http.ResponseWriter, r *http.Request) {
    fmt.Fprintf(w, "Hello, %s!", r.URL.Path[1:])
}

func main() {
    // 简单路由
    http.HandleFunc("/hello/", helloHandler)
    
    // 静态文件服务
    fs := http.FileServer(http.Dir("static/"))
    http.Handle("/static/", http.StripPrefix("/static/", fs))
    
    // 启动服务器
    fmt.Println("Server starting on port 8080...")
    err := http.ListenAndServe(":8080", nil)
    if err != nil {
        panic(err)
    }
}

3.2 HTTP 客户端

package main

import (
    "fmt"
    "io"
    "net/http"
    "os"
)

func main() {
    // 简单GET请求
    resp, err := http.Get("https://jsonplaceholder.typicode.com/posts/1")
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()
    
    body, err := io.ReadAll(resp.Body)
    if err != nil {
        panic(err)
    }
    fmt.Println("Response:", string(body))

    // POST请求
    reqBody := strings.NewReader(`{"title":"foo","body":"bar","userId":1}`)
    resp, err = http.Post("https://jsonplaceholder.typicode.com/posts", 
        "application/json", reqBody)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()
    
    body, err = io.ReadAll(resp.Body)
    fmt.Println("POST Response:", string(body))

    // 更复杂的请求
    client := &http.Client{}
    req, err := http.NewRequest("PUT", "https://jsonplaceholder.typicode.com/posts/1", 
        strings.NewReader(`{"id":1,"title":"foo","body":"bar","userId":1}`))
    if err != nil {
        panic(err)
    }
    req.Header.Add("Content-Type", "application/json")
    req.Header.Add("Authorization", "Bearer token123")
    
    resp, err = client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()
    
    body, err = io.ReadAll(resp.Body)
    fmt.Println("PUT Response:", string(body))
}

4. encoding/json 和 encoding/xml 包

4.1 JSON 处理

package main

import (
    "encoding/json"
    "fmt"
    "log"
)

type Person struct {
    Name    string `json:"name"`
    Age     int    `json:"age"`
    Address string `json:"address,omitempty"`
}

func main() {
    // 编码为JSON
    p := Person{Name: "Alice", Age: 25}
    jsonData, err := json.Marshal(p)
    if err != nil {
        log.Fatal(err)
    }
    fmt.Println("JSON:", string(jsonData))

    // 美化输出
    jsonData, err = json.MarshalIndent(p, "", "  ")
    if err != nil {
        log.Fatal(err)
    }
    fmt.Println("Pretty JSON:\n", string(jsonData))

    // 解码JSON
    jsonStr := `{"name":"Bob","age":30}`
    var p2 Person
    err = json.Unmarshal([]byte(jsonStr), &p2)
    if err != nil {
        log.Fatal(err)
    }
    fmt.Printf("Decoded: %+v\n", p2)

    // 处理未知结构的JSON
    var data map[string]interface{}
    jsonStr = `{"name":"Charlie","age":35,"skills":["Go","Python"]}`
    err = json.Unmarshal([]byte(jsonStr), &data)
    if err != nil {
        log.Fatal(err)
    }
    fmt.Println("Name:", data["name"])
    fmt.Println("Skills:", data["skills"].([]interface{}))
}

5. time 包

package main

import (
    "fmt"
    "time"
)

func main() {
    // 获取当前时间
    now := time.Now()
    fmt.Println("Current time:", now)

    // 格式化时间
    fmt.Println("Formatted:", now.Format("2006-01-02 15:04:05"))
    fmt.Println("RFC3339:", now.Format(time.RFC3339))

    // 解析时间
    t, err := time.Parse("2006-01-02", "2023-05-20")
    if err != nil {
        panic(err)
    }
    fmt.Println("Parsed time:", t)

    // 时间运算
    tomorrow := now.Add(24 * time.Hour)
    fmt.Println("Tomorrow:", tomorrow)

    duration := tomorrow.Sub(now)
    fmt.Println("Duration until tomorrow:", duration)

    // 定时器
    timer := time.NewTimer(2 * time.Second)
    <-timer.C
    fmt.Println("Timer fired")

    // Ticker
    ticker := time.NewTicker(500 * time.Millisecond)
    done := make(chan bool)
    go func() {
        for {
            select {
            case <-done:
                return
            case t := <-ticker.C:
                fmt.Println("Tick at", t)
            }
        }
    }()
    time.Sleep(2 * time.Second)
    ticker.Stop()
    done <- true
    fmt.Println("Ticker stopped")
}

6. sync 包

6.1 互斥锁 (Mutex)

package main

import (
    "fmt"
    "sync"
    "time"
)

type SafeCounter struct {
    mu sync.Mutex
    v  map[string]int
}

func (c *SafeCounter) Inc(key string) {
    c.mu.Lock()
    c.v[key]++
    c.mu.Unlock()
}

func (c *SafeCounter) Value(key string) int {
    c.mu.Lock()
    defer c.mu.Unlock()
    return c.v[key]
}

func main() {
    c := SafeCounter{v: make(map[string]int)}
    for i := 0; i < 1000; i++ {
        go c.Inc("somekey")
    }

    time.Sleep(time.Second)
    fmt.Println(c.Value("somekey"))
}

6.2 读写锁 (RWMutex)

package main

import (
    "fmt"
    "sync"
    "time"
)

type DataStore struct {
    data map[string]string
    mu   sync.RWMutex
}

func (ds *DataStore) Set(key, value string) {
    ds.mu.Lock()
    defer ds.mu.Unlock()
    ds.data[key] = value
}

func (ds *DataStore) Get(key string) string {
    ds.mu.RLock()
    defer ds.mu.RUnlock()
    return ds.data[key]
}

func main() {
    ds := DataStore{data: make(map[string]string)}

    // 写操作
    go func() {
        for i := 0; i < 5; i++ {
            ds.Set(fmt.Sprintf("key%d", i), fmt.Sprintf("value%d", i))
            time.Sleep(100 * time.Millisecond)
        }
    }()

    // 读操作
    for i := 0; i < 10; i++ {
        go func(i int) {
            for j := 0; j < 5; j++ {
                fmt.Printf("Goroutine %d: key%d=%s\n", i, j, ds.Get(fmt.Sprintf("key%d", j)))
                time.Sleep(50 * time.Millisecond)
            }
        }(i)
    }

    time.Sleep(2 * time.Second)
}

6.3 WaitGroup

package main

import (
    "fmt"
    "sync"
    "time"
)

func worker(id int, wg *sync.WaitGroup) {
    defer wg.Done()
    
    fmt.Printf("Worker %d starting\n", id)
    time.Sleep(time.Second)
    fmt.Printf("Worker %d done\n", id)
}

func main() {
    var wg sync.WaitGroup
    
    for i := 1; i <= 5; i++ {
        wg.Add(1)
        go worker(i, &wg)
    }
    
    wg.Wait()
    fmt.Println("All workers completed")
}

6.4 Once

package main

import (
    "fmt"
    "sync"
)

var (
    instance *singleton
    once     sync.Once
)

type singleton struct {
    value int
}

func getInstance() *singleton {
    once.Do(func() {
        instance = &singleton{value: 42}
    })
    return instance
}

func main() {
    var wg sync.WaitGroup
    
    for i := 0; i < 10; i++ {
        wg.Add(1)
        go func() {
            defer wg.Done()
            s := getInstance()
            fmt.Printf("Instance address: %p, value: %d\n", s, s.value)
        }()
    }
    
    wg.Wait()
}

6.5 Cond

package main

import (
    "fmt"
    "sync"
    "time"
)

func main() {
    var m sync.Mutex
    c := sync.NewCond(&m)
    var ready bool

    for i := 0; i < 3; i++ {
        go func(id int) {
            time.Sleep(time.Duration(id) * 100 * time.Millisecond)
            
            m.Lock()
            ready = true
            fmt.Printf("Worker %d is ready\n", id)
            c.Signal() // 通知等待的goroutine
            m.Unlock()
        }(i)
    }

    m.Lock()
    for !ready {
        fmt.Println("Waiting...")
        c.Wait() // 等待条件满足
    }
    m.Unlock()
    
    fmt.Println("All workers are ready")
}

总结

本教程涵盖了Go语言标准库中几个最常用的包：

  1. io/io/ioutil：提供了基本的I/O接口和工具函数
  2. os/exec：用于操作系统功能和执行外部命令
  3. net/http：实现HTTP客户端和服务器功能
  4. encoding/json/xml：处理JSON和XML数据的编码和解码
  5. time：处理时间和日期相关操作
  6. sync：提供并发编程的同步原语，如互斥锁、读写锁等

这些标准库包是Go语言开发中的基础工具，熟练掌握它们将大大提高开发效率和代码质量。
